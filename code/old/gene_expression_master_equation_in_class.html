# our parameters
r = 0.5 # min^-1
gamma = 0.1 # min^-1
dt = 0.1 # min
n_steps = 500

# maximum number of mRNAs that we compute probabilities for
M = 20

# arry to store mRNA probabilites
mRNA = np.zeros([M, n_steps])

# set the probability at t=0
mRNA[0,0] = 1

Loop through time, updating our probability array.

In [62]:
# loop through time
for t in range(1, n_steps): 
    
    # update the mRNA counts for 1 through M-1
    for m in range(1, M-1):
        mRNA[m, t] = mRNA[m, t-1] + r*dt*mRNA[m-1,t-1] + gamma*(m+1)*dt*mRNA[m+1,t-1] + \
                    - r*dt*mRNA[m,t-1] - gamma*m*dt*mRNA[m,t-1]
        
    # left boundary
    mRNA[0, t] = mRNA[0, t-1] + gamma*dt*mRNA[1,t-1] - r*dt*mRNA[0,t-1]

    # right boundary
    mRNA[M-1, t] = mRNA[M-1, t-1] + r*dt*mRNA[M-2,t-1] - gamma*(M-1)*dt*mRNA[M-1,t-1]
