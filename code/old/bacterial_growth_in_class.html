In [39]:
# grab all the fluor files
im_names = np.sort(glob.glob("./data/ecoli_growth/ecoli_TRITC_*.tif"))

# number of frames to be looped through
n_frames = len(im_names)

# initilize an array to store bacterial areas
areas = np.zeros(n_frames)

# loop through frames, calculate area for each
for i in range(n_frames): 
    
    # read in the image
    im = skimage.io.imread(im_names[i])
    
    # normalize the image 
    im_float = (im - np.min(im)) / (np.max(im) - np.min(im))
    
    # apply the threshold 
    im_thresh = im_float > thresh 
    
    # find the bacterial area
    areas[i] = np.sum(im_thresh)

Calculate the error for a range of $k$ values to find the best fit.

In [52]:
def error(data_curve, theory_curve):
    """
    This error function returns the sum of squared errors
    between the theory curve and the data curve. 
    """
    
    # compute the error
    error = np.sum((data_curve - theory_curve)**2)
    
    return error
