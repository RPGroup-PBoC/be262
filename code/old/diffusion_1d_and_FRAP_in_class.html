# jumping rate
k = 1 # sec^-1

# number of positions the particles can live in
n_boxes = 40

# number of time steps
n_steps = 100

# time step
dt = 0.1 # sec

# initialize an array to store probabilities
p = np.zeros([n_boxes, n_steps])

# find the center box
n_center = int(n_boxes/2)

# set the center to have probability 1 at first time step
p[n_center,0] = 1

Loop through time, updating the probabilities.

In [11]:
# first loop through time
for t in range(1, n_steps):
    
    # then loop through all internal positions
    for n in range(1,n_boxes-1): 
        p[n,t] = p[n,t-1] + k*dt*p[n-1,t-1] + k*dt*p[n+1,t-1] - 2*k*dt*p[n,t-1]
        
    # edge case for position 0
    p[0,t] = p[0,t-1] + k*dt*p[1,t-1] - k*dt*p[0,t-1]
    
    # edge for position -1 
    p[-1,t] = p[-1,t-1] + k*dt*p[-2,t-1] - k*dt*p[-1,t-1]

Set up parameters for FRAP "experiment." The only thing that changes is the initial condition of the probabilities.

In [19]:
# allow enough time for recovery to happen
n_steps = 1000

# initialize a probability array that has been photobleach
p = np.zeros([n_boxes, n_steps])
p[:,0] = 1

# assign boxes to photobleach
n_bleach = 10
start = n_center - int(n_bleach/2)
end = n_center + int(n_bleach/2)
p[start:end,0] = 0

# normalize so probability sums to 1
p[:,0] = p[:,0] / np.sum(p[:,0])
