Simulate a walker taking steps right (+1) and left (-1) for 1000 steps and plot the resulting trajectory.

In [4]:
# number of steps
n_steps = 1000

# array to store walker positions
positions = np.empty(n_steps)

# initilize the walker's first postion
positions[0] = 0

# loop positions array, stepping left or right each time
for t in range(1, n_steps):
    
    # produce our random number
    rand = np.random.uniform()
    
    # step right
    if rand > 0.5:
        positions[t] = positions[t-1] + 1
        
    # step left
    else:
        positions[t] = positions[t-1] - 1

Now repeat this process for 1000 walkers, and plot the results.

In [6]:
# number of trajectories
n_traj = 1000

# initilize 2D array for trajectories
positions_2D = np.empty([n_traj, n_steps])

# initialize all walker to be a position 0
positions_2D[:,0] = 0 

# first, loop through trajectories
for i in range(n_traj):
    
    # next, loop through the time steps
    for j in range(1, n_steps):
        
        # generate random number
        rand = np.random.uniform()
        
        # step right
        if rand > 0.5:
            positions_2D[i,j] = positions_2D[i,j-1] + 1
    
        # step left
        else:
            positions_2D[i,j] = positions_2D[i,j-1] - 1
