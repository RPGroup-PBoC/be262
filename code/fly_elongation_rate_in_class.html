def spot_intensity(file_name_list):
    """
    This function takes in a list of files names and returns
    the spot inensity over time as an array
    """
    
    # number of frames to loop through
    n_frames = len(file_name_list)
    
    # array to store intensities
    intensities = np.zeros(n_frames)
    
    # loop through the frames of the movie
    for i in range(n_frames):
        
        # read in the image
        im = skimage.io.imread(file_name_list[i])
        
        # flatten the image into a 1D array
        im_sorted = np.sort(im.flatten())
        
        # reverse the order of the sorted
        # im_reversed = np.flip(im_sorted)
        
        # calculate the intensity of the spot
        # intensities[i] = np.sum(im_reversed[0:5])
        intensities[i] = np.sum(im_sorted[-5:]) # alternative way to get top 5 
        
    return intensities
