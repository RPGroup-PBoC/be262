# number of vials 
n_vials = 107

# number of flies in each vial
n_flies = 16

# number of generations to run the simulation
n_generations = 20

# array to store allele frequencies
red_freq = np.zeros([n_vials, n_generations])

# initialize all vials to have allele freq 0.5 at generation 0
red_freq[:,0] = 0.5

Update our frequency array by looping through generations for each vial

In [15]:
# loop through the number of vials
for v in range(n_vials):
    
    # loop through the number of generations
    for g in range(1, n_generations):
        
        # determine allele freq of red in previous generation
        old_red_freq = red_freq[v, g-1]
        
        # simulate the next generations (treating pop as bag alleles)
        # 0 is white allele, 1 is red allele
        new_alleles = np.random.choice([0,1], size=n_flies*2, 
                                      p=[1-old_red_freq, old_red_freq])
        
        # calculating new red allele frequency
        new_red_freq = sum(new_alleles) / len(new_alleles)
        
        # udpdate our red frequency array
        red_freq[v, g] = new_red_freq
